\begin{algorithm}
\caption {BUBBLE SORT}
\begin{algorithmic}[1]
\Procedure{Bubble Sort}{A,n}
\For{$i  \gets 0$  to $n-1$}
	\For{$ j \gets 0$ to $n-1-i$}
		\If($A[j]>A[j+1]$}
			\State swap $a[j] $ and $a[j+1]$
		\EndIf
	\EndFor
\EndFor
\EndProcedure

\end{algorithmic}
\end{algorithm}
